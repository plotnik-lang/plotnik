================================================================================
Tree-sitter parenthesized sequence is rejected
:error
================================================================================
Q = (program ((a) (b)))
--------------------------------------------------------------------------------

================================================================================
Deprecated bang negation is rejected
:error
================================================================================
Q = (f !name)
--------------------------------------------------------------------------------

================================================================================
Lowercase definition name is rejected
:error
================================================================================
q = (a)
--------------------------------------------------------------------------------

================================================================================
Bare identifier is not a pattern
:error
================================================================================
Q = foo
--------------------------------------------------------------------------------

================================================================================
Double quantifier and double capture are rejected
:error
================================================================================
A = (p (a)*+)
B = (p (a) @x @y)
--------------------------------------------------------------------------------

================================================================================
Capture with dotted name is rejected
:error
================================================================================
Q = (p (a) @foo.bar)
--------------------------------------------------------------------------------

================================================================================
PascalCase capture is rejected
:error
================================================================================
Q = (p (a) @Foo)
--------------------------------------------------------------------------------

================================================================================
Capture type on discard is rejected
:error
================================================================================
Q = (p (a) @_x :: T)
--------------------------------------------------------------------------------

================================================================================
Capture type without a capture is rejected
:error
================================================================================
Text = (identifier) :: str
--------------------------------------------------------------------------------

================================================================================
Single-colon capture type is rejected
:error
================================================================================
Q = (p (a) @x : T)
--------------------------------------------------------------------------------

================================================================================
Anchor directly in alternation is rejected
:error
================================================================================
Q = (p [(a) . (b)])
--------------------------------------------------------------------------------

================================================================================
Lowercase alternative label is rejected
:error
================================================================================
Q = (p [name: (x)])
--------------------------------------------------------------------------------

================================================================================
Anchor or negated field as definition body is rejected
:error
================================================================================
A = .
B = -name
--------------------------------------------------------------------------------

================================================================================
Anchor and negated field as field values are rejected
:error
================================================================================
A = (p f: .)
B = (p f: -g)
--------------------------------------------------------------------------------

================================================================================
Suffix on anchor or negated field is rejected
:error
================================================================================
A = (p . *)
B = (p -f @x)
--------------------------------------------------------------------------------

================================================================================
Negated field in a sequence is rejected
:error
================================================================================
Q = (p {(a) -absent})
--------------------------------------------------------------------------------

================================================================================
Anchor or negated field as a labeled alternative body is rejected
:error
================================================================================
A = (p [X: . Y: (a)])
B = (p [X: -absent Y: (a)])
--------------------------------------------------------------------------------

================================================================================
Negated field as an unlabeled alternative is rejected
:error
================================================================================
Q = (p [(a) -absent])
--------------------------------------------------------------------------------

================================================================================
Children in MISSING are rejected
:error
================================================================================
A = (MISSING (something))
B = (MISSING name: (x))
--------------------------------------------------------------------------------

================================================================================
Anchor or negated field in MISSING is rejected
:error
================================================================================
A = (MISSING .)
B = (MISSING -name)
--------------------------------------------------------------------------------

================================================================================
Second MISSING argument is rejected
:error
================================================================================
Q = (MISSING identifier extra)
--------------------------------------------------------------------------------

================================================================================
Tree-sitter predicates are rejected
:error
================================================================================
Q = ((identifier) @x (#eq? @x "foo"))
--------------------------------------------------------------------------------

================================================================================
Comma and pipe separators are rejected
:error
================================================================================
A = (p (a), (b))
B = (p [(a) | (b)])
--------------------------------------------------------------------------------

================================================================================
Reference with children is rejected
:error
================================================================================
Q = (Foo (a))
--------------------------------------------------------------------------------

================================================================================
Slash refinement without subtype is rejected
:error
================================================================================
Q = (expression/)
--------------------------------------------------------------------------------

================================================================================
Loose supertype separator is rejected
:error
================================================================================
Q = (a # b)
--------------------------------------------------------------------------------

================================================================================
Shebang after the first line is rejected
:error
================================================================================
Q = (a)
#!/usr/bin/env x
--------------------------------------------------------------------------------

================================================================================
Unterminated string is rejected
:error
================================================================================
Q = (p "abc)
R = (r)
--------------------------------------------------------------------------------

================================================================================
Block comment with star before closing is not a comment
:error
================================================================================
Q = (a) /* x **/
--------------------------------------------------------------------------------
