Q = (program
  {
    (lexical_declaration) @decl
    (debugger_statement) @dbg
  }? @attrs
)
------------------- TYPESCRIPT -------------------
export interface QAttrs {
  dbg: Node;
  decl: Node;
}

export interface Q {
  attrs: QAttrs | null;
}
---------------------- RUST ----------------------
use ::plotnik_rt::Node;

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct Q<'t> {
    pub attrs: ::core::option::Option<QAttrs<'t>>,
}

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct QAttrs<'t> {
    pub decl: Node<'t>,
    pub dbg: Node<'t>,
}
