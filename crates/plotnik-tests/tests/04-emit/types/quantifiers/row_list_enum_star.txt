use ::plotnik_rt::Node;

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct Q<'t> {
    pub rows: ::std::vec::Vec<QRows<'t>>,
}

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum QRows<'t> {
    A { id: Node<'t> },
    B { n: Node<'t> },
}
