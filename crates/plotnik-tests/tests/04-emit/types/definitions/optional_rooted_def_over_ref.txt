---------------------- RUST ----------------------
use ::plotnik_rt::Node;

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct Row<'t> {
    pub key: Node<'t>,
    pub val: Node<'t>,
}

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct Q<'t> {
    pub first: ::core::option::Option<Row<'t>>,
}

pub type MaybeRow<'t> = ::core::option::Option<Row<'t>>;
