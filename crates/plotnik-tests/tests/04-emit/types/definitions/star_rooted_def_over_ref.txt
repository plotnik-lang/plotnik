use ::plotnik_rt::Node;

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct Row<'t> {
    pub key: Node<'t>,
    pub val: Node<'t>,
}

pub type Rows<'t> = ::std::vec::Vec<Row<'t>>;

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct Q<'t> {
    pub rows: ::std::vec::Vec<Row<'t>>,
}
