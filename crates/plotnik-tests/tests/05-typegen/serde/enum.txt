Q = [
  Str: (string) @s
  Num: (number) @n
]
------------------- TYPESCRIPT -------------------
export type Q =
  | { $tag: "Str"; $data: { s: Node } }
  | { $tag: "Num"; $data: { n: Node } };
---------------------- RUST ----------------------
use ::plotnik_rt::Node;

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum Q<'t> {
    Str { s: Node<'t> },
    Num { n: Node<'t> },
}

impl ::plotnik_rt::SerializeWithSource for Q<'_> {
    fn serialize_with_source<S>(
        &self,
        source: &str,
        serializer: S,
    ) -> ::core::result::Result<S::Ok, S::Error>
    where
        S: ::plotnik_rt::serde::Serializer,
    {
        use ::plotnik_rt::serde::ser::SerializeMap as _;

        match self {
            Q::Str { s: v0 } => {
                struct Data<'a, 't> {
                    v0: &'a Node<'t>,
                    source: &'a str,
                }
                impl ::plotnik_rt::serde::Serialize for Data<'_, '_> {
                    fn serialize<S>(
                        &self,
                        serializer: S,
                    ) -> ::core::result::Result<S::Ok, S::Error>
                    where
                        S: ::plotnik_rt::serde::Serializer,
                    {
                        use ::plotnik_rt::serde::ser::SerializeMap as _;

                        let mut map = serializer.serialize_map(Some(1))?;
                        map.serialize_entry("s", &::plotnik_rt::WithSource::new(self.v0, self.source))?;
                        map.end()
                    }
                }

                let mut map = serializer.serialize_map(Some(2))?;
                map.serialize_entry("$tag", "Str")?;
                map.serialize_entry("$data", &Data { v0, source })?;
                map.end()
            }
            Q::Num { n: v0 } => {
                struct Data<'a, 't> {
                    v0: &'a Node<'t>,
                    source: &'a str,
                }
                impl ::plotnik_rt::serde::Serialize for Data<'_, '_> {
                    fn serialize<S>(
                        &self,
                        serializer: S,
                    ) -> ::core::result::Result<S::Ok, S::Error>
                    where
                        S: ::plotnik_rt::serde::Serializer,
                    {
                        use ::plotnik_rt::serde::ser::SerializeMap as _;

                        let mut map = serializer.serialize_map(Some(1))?;
                        map.serialize_entry("n", &::plotnik_rt::WithSource::new(self.v0, self.source))?;
                        map.end()
                    }
                }

                let mut map = serializer.serialize_map(Some(2))?;
                map.serialize_entry("$tag", "Num")?;
                map.serialize_entry("$data", &Data { v0, source })?;
                map.end()
            }
        }
    }
}
