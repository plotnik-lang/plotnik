Q = {
  (identifier) @k
  (number) @v
}* @rows
------------------- TYPESCRIPT -------------------
export interface QRows {
  k: Node;
  v: Node;
}

export interface Q {
  rows: QRows[];
}
---------------------- RUST ----------------------
use ::plotnik_rt::Node;

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct Q<'t> {
    pub rows: ::std::vec::Vec<QRows<'t>>,
}

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct QRows<'t> {
    pub k: Node<'t>,
    pub v: Node<'t>,
}

impl ::plotnik_rt::SerializeWithSource for Q<'_> {
    fn serialize_with_source<S>(
        &self,
        source: &str,
        serializer: S,
    ) -> ::core::result::Result<S::Ok, S::Error>
    where
        S: ::plotnik_rt::serde::Serializer,
    {
        use ::plotnik_rt::serde::ser::SerializeMap as _;

        let mut map = serializer.serialize_map(Some(1))?;
        map.serialize_entry("rows", &::plotnik_rt::WithSource::new(&self.rows, source))?;
        map.end()
    }
}

impl ::plotnik_rt::SerializeWithSource for QRows<'_> {
    fn serialize_with_source<S>(
        &self,
        source: &str,
        serializer: S,
    ) -> ::core::result::Result<S::Ok, S::Error>
    where
        S: ::plotnik_rt::serde::Serializer,
    {
        use ::plotnik_rt::serde::ser::SerializeMap as _;

        let mut map = serializer.serialize_map(Some(2))?;
        map.serialize_entry("k", &::plotnik_rt::WithSource::new(&self.k, source))?;
        map.serialize_entry("v", &::plotnik_rt::WithSource::new(&self.v, source))?;
        map.end()
    }
}
