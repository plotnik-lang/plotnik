Q = (program [
  Stmt: (expression_statement [
    Yes: (identifier)
    No: (number)
  ] @kind)
] @item)
------------------- TYPESCRIPT -------------------
export type QItem =
  | { $tag: "Stmt"; $data: { kind: QItemStmtKind } };

export interface Q {
  item: QItem;
}

export type QItemStmtKind =
  | { $tag: "Yes" }
  | { $tag: "No" };
---------------------- RUST ----------------------
#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct Q {
    pub item: QItem,
}

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum QItem {
    Stmt { kind: QItemStmtKind },
}

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum QItemStmtKind {
    Yes,
    No,
}
