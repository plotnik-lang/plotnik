use ::plotnik_rt::Node;

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum Entry<'t> {
    Id { id: Node<'t> },
    Num { num: Node<'t> },
}

pub type Items<'t> = ::std::vec::Vec<Entry<'t>>;

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct Q<'t> {
    pub items: ::std::vec::Vec<Entry<'t>>,
}
