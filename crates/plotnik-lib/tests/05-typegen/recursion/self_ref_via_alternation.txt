Expr = [
  Lit: (number) @value
  Binary: (binary_expression
    left: (Expr) @left
    right: (Expr) @right
  )
]
==== types ====
export interface ExprLit {
  $tag: "Lit";
  $data: { value: Node };
}

export interface ExprBinary {
  $tag: "Binary";
  $data: { left: Expr; right: Expr };
}

export type Expr = ExprLit | ExprBinary;
